export const ConvertedColor = () => {
    return ConvertedColors.map((colorGroup, i) => (
        <div key={`colorGroupConverted-${i}`}>
            <h2>{colorGroup.title}</h2>
            {colorGroup.colors.map((color, j) => (
                <ColorSingle
                    colorHex={color.hex}
                    colorName={color.name}
                    colorUse={color.use}
                    colorForeground={color.foreground || settingColors.foreground}
                    colorBackground={color.background || settingColors.background}
                    key={`colorGroupConverted-${i}-${j}`}
                />
            ))}
        </div>
    ))
}
